{{~ # --- 世界情報整理 --- ~}}
{{~

  map_info = {}
  # グローバルなtipsのリスト
  tips_list = []
  # ポーンがマップ上に生存しているか
  is_on_map = false
  if pawn.Map != null
    is_on_map = true
  end

  # ■Pawn.Mapがnullでなければ実行
  if is_on_map
    # 現在マップの所持派閥
    map_parent_faction = pawn.Map.ParentFaction
    # 現在マップのバイオーム情報
    map_biome = pawn.Map.Biome.label
    # 現在マップの気温
    floor_temperature = pawn.Map.mapTemperature.OutdoorTemp | math.floor
    # 経過tick数
    game_ticks_elapsed = Find.tickmanager.ticksgame
    
    # 異常な温度の場合のみ整形して表示
    if floor_temperature >= -74 && (floor_temperature >= 35 || floor_temperature <= 0)
      map_temperature = "外気温:" + floor_temperature + "C"
    end
    
    # 現在マップの天気、空間
    for item in keyword_environment
      if pawn.Map.weather | string.contains item.key
        map_environment = item.value
      end
    end

    # 現在マップのマップイベント
    for item in keyword_map_condition
      if pawn.Map.GameConditionManager.ActiveConditions != null
        for condition in pawn.Map.GameConditionManager.ActiveConditions
          if condition.label | string.contains item.key
            map_conditions = map_conditions | array.add item.value
          end
        end
      end
    end

    # インシデントの内容取得
    if prompt | string.contains "Talk "
      map_incident_all = prompt | string.split "[" | array.last | string.split "]" | array.first
      map_incident = map_incident_all | string.split "\n" | array.first
    end

    # プレイヤー派閥名
    player_faction_name = Find.FactionManager.OfPlayer.Name
    # プレイヤー派閥の人型ポーン
    player_humanlike_total_members = pawn.Map.mapPawns.FreeColonistsSpawned
    
    # プレイヤー派閥の人型ポーン取得
    player_humanlike_downed_members = []
    player_humanlike_active_members = []
    for humanlike_pawn in player_humanlike_total_members
      if humanlike_pawn.Downed
        # プレイヤー派閥のダウンした人型ポーン
        player_humanlike_downed_members = player_humanlike_downed_members | array.add humanlike_pawn
      else
        # プレイヤー派閥の行動可能人型ポーン
        player_humanlike_active_members = player_humanlike_active_members | array.add humanlike_pawn
      end
    end

    

    # 死亡ポーンの中から自派閥のポーンを取得
    dead_player_pawns = []
    for target_pawn in dead_pawns
      if target_pawn.isColonist
        dead_player_pawns = dead_player_pawns | array.add target_pawn
      end
    end


    # 危険判定
    danger_flag = false
    if (pawn.map.dangerWatcher.DangerRating | string.contains "High") || (pawn.map.dangerWatcher.DangerRating | string.contains "Low")
      danger_flag = true
    end

    # ヒト狩り判定
    displayed_manhunters = []
    if danger_flag
      for map_pawn in pawn.Map.mapPawns.AllPawnsSpawned
        if map_pawn.InMentalState
          if map_pawn.MentalStateDef == "ManhunterPermanent" || map_pawn.MentalStateDef == "Manhunter"
            if !(displayed_manhunters | array.contains map_pawn.name)
              manhunter_animals = manhunter_animals | array.add map_pawn.name
              displayed_manhunters = displayed_manhunters | array.add map_pawn.name
            end
          end
        end
      end
    end

    # マップ内の外部ポーングループ情報整理
    lords = []
    lords_dict = {}  # faction文字列をキーにした集計用辞書

    if pawn.Map.lordmanager.lords | array.size > 0
      
      has_hostile_lord = false

      for lord in pawn.map.lordmanager.lords

        # 除外チェック
        is_exclude_lordjob = false
        for exclude_lordjob in keyword_exclude_lordjob
          if lord.LordJob | string.contains exclude_lordjob
            is_exclude_lordjob = true
          end
        end

        # 全滅チェック
        is_alive_lord = false
        if lord.numPawnsEverGained != null && lord.numPawnsLostViolently != null
          if (lord.numPawnsEverGained - lord.numPawnsLostViolently) > 0
            is_alive_lord = true
          end
        end

        if lord.AnyActivePawn && !is_exclude_lordjob && is_alive_lord

          if lord.faction != null
            faction_key = lord.faction
          else
            faction_key = "未所属"
          end
          # 特殊グループ（護衛対象）ではないか
          escort_target_name = null
          escort_text = null
          for item in keyword_extra_lordjob
            if lord.LordJob | string.contains item.key
              if lord.ownedPawns != null
                for lord_pawn in lord.ownedPawns
                  if lord_pawn.lord != null
                    if lord_pawn.lord.LordJob | string.contains item.key
                      escort_target_name = escort_target_name | array.add lord_pawn.name
                      escort_text = item.value
                    end
                  end
                end
              end
            end
          end

          # ▼ 初回登場派閥なら辞書に新規登録
          if lords_dict[faction_key] == null

            lord_job = null
            lord_toil = null
            lord_trader_kind = null
            enemy_lord_flag = false

            if lord.faction != null && !lord.Faction.isPlayer
              if lord.Faction.PlayerRelationKind == "Hostile"
                faction_relation_kind = "敵対派閥:"
              else if lord.Faction.PlayerRelationKind == "Ally"
                faction_relation_kind = "同盟派閥:"
              else if lord.Faction.PlayerRelationKind == "Neutral"
                faction_relation_kind = "中立派閥:"
              else
                faction_relation_kind = ""
              end
            end

            for item in keyword_lordjob
              if lord.LordJob | string.contains item.key
                lord_job = item.value
              end
            end

            for item in keyword_lordtoil
              if lord.CurLordToil | string.contains item.key
                lord_toil = item.value
              end
            end
            
            for lord_pawn in lord.ownedPawns
              if enemy_lord_flag == false
                enemy_lord_flag = lord_pawn.profile | string.contains "Role: Enemy Group"
                if enemy_lord_flag
                  has_hostile_lord = true
                end
              end
              if lord_trader_kind == null
                if lord_pawn.trader != null
                  lord_trader_kind = lord_pawn.trader.traderKind.label
                end
              end
            end


            lords_dict[faction_key] = {
              "faction":               lord.faction,
              "faction_relation_kind": faction_relation_kind,
              "lord_job":              lord_job,
              "lord_toil":             lord_toil,
              "trader_kind":           lord_trader_kind,
              "initial_members":       lord.numPawnsEverGained,
              "downed_members":        lord.numPawnsLostViolently,
              "active_members":        lord.numPawnsEverGained - lord.numPawnsLostViolently,
              "enemy_lord_flag":       enemy_lord_flag,
              "has_hostile_lord":      has_hostile_lord,
              "escort_target_name":    escort_target_name,
              "escort_text":           escort_text,
            }

          else
            # ▼ 既存派閥なら該当派閥キーの内容を編集、絶対他にいい方法ある
            lords_dict[faction_key].initial_members = lords_dict[faction_key].initial_members + lord.numPawnsEverGained
            lords_dict[faction_key].downed_members  = lords_dict[faction_key].downed_members  + lord.numPawnsLostViolently
            lords_dict[faction_key].active_members  = lords_dict[faction_key].active_members  + (lord.numPawnsEverGained - lord.numPawnsLostViolently)
            lords_dict[faction_key].escort_target_name = escort_target_name
            lords_dict[faction_key].escort_text = escort_text
          end
        end
      end

        # ▼ 辞書から配列に変換して出力用lordsを作成
        for item in lords_dict
          lords = lords | array.add item.value
        end
    end
    
    map_info = {
      "is_on_map": is_on_map, #マップが存在するか
      "parent_faction": map_parent_faction,  #現在マップの親派閥
      "biome": map_biome,  #バイオーム
      "temperature": map_temperature,  #外気温
      "environment": map_environment,  #天気、場所
      "conditions": map_conditions, #マップイベント
      "incident": map_incident,  #事件
      "game_ticks_elapsed": game_ticks_elapsed, #経過tick数
      "player_faction_name": player_faction_name,  #プレイヤー派閥名
      "player_humanlike_total_members": player_humanlike_total_members,  #プレイヤー派閥の人型ポーン数
