{{~

  # 初期化
  danger_flag = false
  if pawn.Map.dangerWatcher.DangerRating == "High" || pawn.Map.dangerWatcher.DangerRating == "Low"
    danger_flag = true
  end

  # ポーン毎の戦闘情報を取得するためポーンループ
  for p in pawns
    if p != null
      
      is_combat = false
      is_drafted = false
      latest_harm_tick = null
      latest_attack_tick = null
      latest_combat_tick = null
      last_combat_tick = null

      # 派閥を持っているか
      has_faction = false
      if p.Faction != null
        has_faction = true
      end

      if has_faction

        # 取得ポーンの派閥を確認
        if p.Faction.IsPlayer
          player_faction = "プレイヤー派閥"
          player_pawns = player_pawns | array.add p.name
        else if p.Faction.PlayerRelationKind == "Hostile"
          hostile_faction = "敵対派閥"
          hostile_pawns = hostile_pawns | array.add p.name
        else if p.Faction.PlayerRelationKind == "Ally"
          ally_faction = "同盟派閥"
          ally_pawns = ally_pawns | array.add p.name
        else if p.Faction.PlayerRelationKind == "Neutral"
          neutral_faction = "中立派閥"
          neutral_pawns = neutral_pawns | array.add p.name
        else
          no_faction = "無所属"
          other_pawns = other_pawns | array.add p.name
        end

        # 狩りの場合でも反応するので注意、プレイヤー派閥の場合のみ取得
        if p.Faction.IsPlayer

          
          # 戦闘中か？danger_flagがtrueの時のみ実行（狩り中でも反応してしまうため）
          if danger_flag

            # 敵は0が初期値、味方は-99999が初期値
            if p.mindState.lastRangedHarmTick != 0 && p.mindState.lastRangedHarmTick != -99999
              latest_harm_tick = p.mindState.lastRangedHarmTick
            end

            # -99999が初期値
            if p.mindState.lastMeleeThreatHarmTick != -99999
              # latest_harm_tickがnullでないなら比較して大きい方で更新
              if latest_harm_tick
                if latest_harm_tick < p.mindState.lastMeleeThreatHarmTick
                  latest_harm_tick = p.mindState.lastMeleeThreatHarmTick
                end
              else
                latest_harm_tick = p.mindState.lastMeleeThreatHarmTick
              end
            end

            # -99999が初期値
            if p.mindState.lastAttackTargetTick != -99999
              latest_attack_tick = p.mindState.lastAttackTargetTick
            end

            # RangeもMeleeもnullでないなら比較、片方がnullなら存在する方でlast_combat_tickを更新
            if latest_harm_tick != null && latest_attack_tick != null
              if latest_harm_tick > latest_attack_tick
                latest_combat_tick = latest_attack_tick
              else
                latest_combat_tick = latest_harm_tick
              end
            else if latest_harm_tick != null
              latest_combat_tick = latest_harm_tick
            else if latest_attack_tick != null
              latest_combat_tick = latest_attack_tick
            end


            # ゲーム内で2時間以内に攻撃被弾していた場合
            if latest_combat_tick != null
              last_combat_tick = Find.tickmanager.ticksgame - latest_combat_tick
              
              if last_combat_tick < 5000
                
                is_combat = true
                combat_text = "現在戦闘中です！"
              else if last_combat_tick < 50000
                is_combat = true
                combat_text = "戦闘状態を継続中です"
              else
                is_combat = false
                combat_text = "非戦闘状態です"
              end
            end
          end
        end
      end

      # 周囲が危険 + 戦闘中の場合
      if danger_flag && is_combat
        combat_pawns = combat_pawns | array.add p.name
      end

    end
  end

  if player_pawns | array.size > 0
    player_pawns_name = player_pawns | array.join ","
    p_text = "プレイヤー派閥のメンバー【" + player_pawns_name + "】"
    text_list = text_list | array.add p_text
  end

  if hostile_pawns | array.size > 0
    hostile_pawns_name = hostile_pawns | array.join ","
    h_text = "敵対派閥のメンバー【" + hostile_pawns_name + "】"
    text_list = text_list | array.add h_text
  end

  if ally_pawns | array.size > 0
    ally_pawns_name = ally_pawns | array.join ","
    a_text = "同盟派閥のメンバー【" + ally_pawns_name + "】"
    text_list = text_list | array.add a_text
  end

  if neutral_pawns | array.size > 0
    neutral_pawns_name = neutral_pawns | array.join ","
    n_text = "中立派閥のメンバー【" + neutral_pawns_name + "】"
    text_list = text_list | array.add n_text
  end

  if other_pawns | array.size > 0
    other_pawns_name = other_pawns | array.join ","
    o_text = "無所属のメンバー【" + other_pawns_name + "】"
    text_list = text_list | array.add o_text
  end

  formatted_combat_pawns = combat_pawns | array.join "、"
  formatted_text_list = text_list | array.join "、"

~}}
